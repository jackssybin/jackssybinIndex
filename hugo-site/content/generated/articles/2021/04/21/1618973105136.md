<p>@ConditionalOnExpression 根据表达式选择性加载
@ConditionalOnProperty  根据配置选择性加载</p>
<pre><code>#消费者总开关，0关1开
mq.cumsumer.enabled=1
#rocketmq消费者开关，true开启，false关闭
rocketmq.comsumer.enabled=false
#rabbitmq消费者开关，true开启，false关闭
rabbitmq.comsumer.enabled=true
#消费者选择
mq.comsumer=rabbitmq
</code></pre>
<p>//布尔值和数字
@Component
@RabbitListener(queues = &quot;monitorDataQueue&quot;)
@ConditionalOnExpression(&quot;${mq.cumsumer.enabled:0}==1&amp;&amp;${rabbitmq.comsumer.enabled:false}&quot;)</p>
<p>//字符串
@Component
@RabbitListener(queues = &quot;monitorDataQueue&quot;)
@ConditionalOnExpression(&quot;'${mq.comsumer}'.equals('rabbitmq')&quot;)</p>
<p>// 直接使用boolean 的校验
// spring.batch.job.enabled = true</p>
<p>@ConditionalOnProperty(prefix = &quot;spring.batch.job&quot;, name = &quot;enabled&quot;, havingValue = &quot;true&quot;, matchIfMissing = true)</p>
