<h2>1.LocalDateTime时间转换</h2>
<pre><code>import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
​
public class DateTimeConversionExample {
    public static void main(String[] args) {
        // 获取当前日期时间
        LocalDateTime now = LocalDateTime.now();

        // 方法1：使用预定义格式器
        DateTimeFormatter isoFormatter = DateTimeFormatter.ISO_LOCAL_DATE_TIME;
        String isoString = now.format(isoFormatter);
        System.out.println(&quot;ISO格式: &quot; + isoString);

        // 方法2：自定义格式
        DateTimeFormatter customFormatter = DateTimeFormatter.ofPattern(&quot;yyyy-MM-dd HH:mm:ss&quot;);
        String customString = now.format(customFormatter);
        System.out.println(&quot;自定义格式: &quot; + customString);

        // 方法3：带本地化格式
        DateTimeFormatter localizedFormatter = DateTimeFormatter.ofPattern(&quot;EEEE, MMMM dd, yyyy hh:mm:ss a&quot;);
        String localizedString = now.format(localizedFormatter);
        System.out.println(&quot;本地化格式: &quot; + localizedString);

        // 方法4：使用预定义常量
        DateTimeFormatter shortFormatter = DateTimeFormatter.ofLocalizedDateTime(
            java.time.format.FormatStyle.SHORT
        );
        String shortString = now.format(shortFormatter);
        System.out.println(&quot;短格式: &quot; + shortString);

        // 方法5：带时区信息（转换为ZonedDateTime）
        String zonedString = now.atZone(java.time.ZoneId.systemDefault())
                               .format(DateTimeFormatter.ISO_ZONED_DATE_TIME);
        System.out.println(&quot;带时区格式: &quot; + zonedString);
    }
}

</code></pre>
<h2>2.DateTimeUtils操作时间</h2>
<pre><code>import java.time.Duration;
import java.time.LocalDateTime;
import java.time.temporal.ChronoUnit;
​
public class DateTimeUtils {
​
    /**
     * 时间加法
     * @param baseTime 基准时间
     * @param amount 要添加的数量
     * @param unit 时间单位
     * @return 计算后的时间
     */
    public static LocalDateTime addTime(LocalDateTime baseTime, long amount, ChronoUnit unit) {
        return baseTime.plus(amount, unit);
    }
​
    /**
     * 时间减法
     * @param baseTime 基准时间
     * @param amount 要减去的数量
     * @param unit 时间单位
     * @return 计算后的时间
     */
    public static LocalDateTime subtractTime(LocalDateTime baseTime, long amount, ChronoUnit unit) {
        return baseTime.minus(amount, unit);
    }
​
    /**
     * 计算两个时间点之间的持续时间（毫秒）
     * @param start 开始时间
     * @param end 结束时间
     * @return 持续时间（毫秒）
     */
    public static long durationInMillis(LocalDateTime start, LocalDateTime end) {
        return Duration.between(start, end).toMillis();
    }
​
    /**
     * 计算两个时间点之间的持续时间（秒）
     * @param start 开始时间
     * @param end 结束时间
     * @return 持续时间（秒）
     */
    public static long durationInSeconds(LocalDateTime start, LocalDateTime end) {
        return Duration.between(start, end).getSeconds();
    }
​
    /**
     * 计算两个时间点之间的持续时间（分钟）
     * @param start 开始时间
     * @param end 结束时间
     * @return 持续时间（分钟）
     */
    public static long durationInMinutes(LocalDateTime start, LocalDateTime end) {
        return Duration.between(start, end).toMinutes();
    }
​
    /**
     * 计算两个时间点之间的持续时间（小时）
     * @param start 开始时间
     * @param end 结束时间
     * @return 持续时间（小时）
     */
    public static long durationInHours(LocalDateTime start, LocalDateTime end) {
        return Duration.between(start, end).toHours();
    }
​
    /**
     * 计算两个时间点之间的持续时间（天）
     * @param start 开始时间
     * @param end 结束时间
     * @return 持续时间（天）
     */
    public static long durationInDays(LocalDateTime start, LocalDateTime end) {
        return ChronoUnit.DAYS.between(start, end);
    }
​
    public static void main(String[] args) {
        // 当前时间
        LocalDateTime now = LocalDateTime.now();
        System.out.println(&quot;当前时间: &quot; + now);

        // 加法示例
        LocalDateTime tomorrow = addTime(now, 1, ChronoUnit.DAYS);
        System.out.println(&quot;明天: &quot; + tomorrow);

        LocalDateTime inTwoHours = addTime(now, 2, ChronoUnit.HOURS);
        System.out.println(&quot;两小时后: &quot; + inTwoHours);

        // 减法示例
        LocalDateTime yesterday = subtractTime(now, 1, ChronoUnit.DAYS);
        System.out.println(&quot;昨天: &quot; + yesterday);

        LocalDateTime thirtyMinutesAgo = subtractTime(now, 30, ChronoUnit.MINUTES);
        System.out.println(&quot;30分钟前: &quot; + thirtyMinutesAgo);

        // 持续时间计算示例
        long millis = durationInMillis(thirtyMinutesAgo, now);
        System.out.println(&quot;30分钟持续时间(毫秒): &quot; + millis);

        long minutes = durationInMinutes(thirtyMinutesAgo, now);
        System.out.println(&quot;30分钟持续时间(分钟): &quot; + minutes);

        // 计算两个特定时间点的持续时间
        LocalDateTime startTime = LocalDateTime.of(2023, 1, 1, 8, 0);
        LocalDateTime endTime = LocalDateTime.of(2023, 1, 1, 10, 30);

        long hoursBetween = durationInHours(startTime, endTime);
        System.out.println(&quot;8:00到10:30之间的持续时间(小时): &quot; + hoursBetween);

        long minutesBetween = durationInMinutes(startTime, endTime);
        System.out.println(&quot;8:00到10:30之间的持续时间(分钟): &quot; + minutesBetween);
    }
}
</code></pre>
<h2>3.完整LocalDateTime使用场景和用法</h2>
<pre><code>import java.time.*;
import java.time.format.*;
import java.time.temporal.*;
import java.util.concurrent.TimeUnit;
​
public class LocalDateTimeTest {
    public static void main(String[] args) {
        // 1. 获取当前日期时间
        LocalDateTime now = LocalDateTime.now();
        System.out.println(&quot;当前时间: &quot; + now);
​
        // 2. 创建指定日期时间
        LocalDateTime specificDate = LocalDateTime.of(2023, Month.OCTOBER, 15, 14, 30);
        System.out.println(&quot;指定日期时间: &quot; + specificDate);
​
        // 3. 日期时间转字符串
        DateTimeFormatter formatter = DateTimeFormatter.ofPattern(&quot;yyyy-MM-dd HH:mm:ss&quot;);
        String formattedDateTime = now.format(formatter);
        System.out.println(&quot;格式化日期时间: &quot; + formattedDateTime);
​
        // 4. 字符串转日期时间
        String dateStr = &quot;2023-12-25 08:30:00&quot;;
        LocalDateTime parsedDateTime = LocalDateTime.parse(dateStr, formatter);
        System.out.println(&quot;解析日期时间: &quot; + parsedDateTime);
​
        // 5. 日期时间加减操作
        LocalDateTime tomorrow = now.plusDays(1);
        LocalDateTime lastWeek = now.minusWeeks(1);
        System.out.println(&quot;明天: &quot; + tomorrow);
        System.out.println(&quot;上周: &quot; + lastWeek);
​
        // 6. 获取时间单位值
        int year = now.getYear();
        Month month = now.getMonth();
        int day = now.getDayOfMonth();
        int hour = now.getHour();
        System.out.printf(&quot;时间单位: %d年 %s月 %d日 %d时%n&quot;, year, month, day, hour);
​
        // 7. 日期时间比较
        boolean isBefore = now.isBefore(specificDate);
        boolean isAfter = now.isAfter(specificDate);
        System.out.println(&quot;当前时间在指定时间之前: &quot; + isBefore);
        System.out.println(&quot;当前时间在指定时间之后: &quot; + isAfter);
​
        // 8. 计算时间间隔
        LocalDateTime start = LocalDateTime.of(2023, 1, 1, 0, 0);
        Duration duration = Duration.between(start, now);
        System.out.println(&quot;间隔天数: &quot; + duration.toDays());
        System.out.println(&quot;间隔小时: &quot; + duration.toHours());
        System.out.println(&quot;间隔分钟: &quot; + duration.toMinutes());
​
        // 9. 时区转换
        ZonedDateTime zonedDateTime = now.atZone(ZoneId.of(&quot;Asia/Shanghai&quot;));
        ZonedDateTime newYorkTime = zonedDateTime.withZoneSameInstant(ZoneId.of(&quot;America/New_York&quot;));
        System.out.println(&quot;上海时间: &quot; + zonedDateTime);
        System.out.println(&quot;纽约时间: &quot; + newYorkTime);
​
        // 10. 与Date互转
        Instant instant = now.atZone(ZoneId.systemDefault()).toInstant();
        java.util.Date legacyDate = java.util.Date.from(instant);
        System.out.println(&quot;传统Date对象: &quot; + legacyDate);
    }
}
</code></pre>
<h3>关键场景说明：</h3>
<ol>
<li><strong>获取当前时间</strong>
<code>LocalDateTime.now()</code> 获取系统当前日期时间</li>
<li><strong>创建指定时间</strong>
<code>LocalDateTime.of()</code> 创建精确到分钟的时间对象</li>
<li><strong>日期时间转字符串</strong>
**使用 **<code>DateTimeFormatter</code> 自定义格式（如 &quot;yyyy-MM-dd HH:mm:ss&quot;）</li>
<li><strong>字符串转日期时间</strong>
<code>LocalDateTime.parse()</code> 配合格式化器解析字符串</li>
<li><strong>日期时间加减</strong>
<code>plusDays()</code>, <code>minusHours()</code> 等方法进行时间运算</li>
<li><strong>获取时间单位</strong>
<code>getYear()</code>, <code>getMonth()</code> 等方法提取时间分量</li>
<li><strong>时间比较</strong>
<code>isBefore()</code>, <code>isAfter()</code> 方法比较时间先后</li>
<li><strong>计算时间间隔</strong>
<code>Duration.between()</code> 计算两个时间的差值</li>
<li><strong>时区转换</strong>
**通过 **<code>atZone()</code> 和 <code>withZoneSameInstant()</code> 处理时区</li>
<li><strong>与传统Date互转</strong>
**通过 **<code>Instant</code> 对象实现与旧版Date的转换</li>
</ol>

<h3>最佳实践（注意事项）：</h3>
<ol>
<li><strong>时区处理</strong>：始终明确时区，避免隐式使用系统默认时区</li>

<li><strong>线程安全</strong>：<code>LocalDateTime</code> 是线程安全的类</li>
<li><strong>格式化重用</strong>：复用 <code>DateTimeFormatter</code> 实例提升性能</li>
