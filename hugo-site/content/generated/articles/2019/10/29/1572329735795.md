<h2>京东喜迎双十一</h2>
<p>这个活动和<code>天猫</code>的活动几乎可以说很像了，都是浏览商家店铺获取金币，
可是这样进度太慢了。于是浏览的大佬的贴子，终于找到可以一键完成养红包的所有任务了。</p>
<h2>使用步骤</h2>
<h3>一、打开谷歌浏览或者火狐也浏览也行（支持开发者模式），按 F12</h3>
<p>进入，切换手机模式如图：
<img src="https://img.hacpai.com/file/2019/10/TIM%E6%88%AA%E5%9B%BE20191029134216-0ae9670a.png?imageView2/2/interlace/1/format/webp" alt="TIM截图20191029134216.png"></p>
<p>738 x 395 1366 x 732</p>
<h3>二、进入京东官网</h3>
<p>1.登入你的京东账号：<a href="http://www.jd.com">http://www.jd.com</a></p>
<p>2.进入活动页面：<a href="https://happy.m.jd.com/babelDiy/GZWVJFLMXBQVEBDQZWMY/XJf8bH6oXDWSgS91daDJzXh9bU7/index.html">https://happy.m.jd.com/babelDiy/GZWVJFLMXBQVEBDQZWMY/XJf8bH6oXDWSgS91daDJzXh9bU7/index.html</a></p>
<h3>三、使用 js 脚本</h3>
<p>把脚本复制到 Console 点击回车就可以看见任务在自动完成
下面是脚本代码：</p>
<pre><code>let productList = [], shopList = [], url = &quot;https://api.m.jd.com/client.action&quot;;
function autoPost(id,type){
        fetch(`${url}?timestamp=${new Date().getTime()}`,{method: &quot;POST&quot;,mode: &quot;cors&quot;,credentials: &quot;include&quot;,headers:{&quot;Content-Type&quot;: &quot;application/x-www-form-urlencoded&quot;},body:`functionId=raisepacket_collectScore&amp;body={&quot;type&quot;:${type},&quot;ext&quot;:&quot;${id}&quot;,&quot;appsign&quot;:1,&quot;msgsign&quot;:2}&amp;client=wh5`})
                .then(function(response){return response.json()})
                .then(function(res){
                        console.log(res.data.biz_msg);
                });
}

function start(){
        fetch(`${url}?${new Date().getTime()}`,{method: &quot;POST&quot;,mode: &quot;cors&quot;,credentials: &quot;include&quot;,headers:{&quot;Content-Type&quot;: &quot;application/x-www-form-urlencoded&quot;},body:'functionId=raisepacket_getShopAndProductList&amp;body=&amp;client=wh5'})
                .then(function(response){return response.json()})
                .then(function(res){
                        productList = res.data.result.productList;
                        shopList  = res.data.result.shopList;
                        console.log(`获取到任务,商品：${productList.length} 商品：${shopList.length}`);
                        autoProductTask();
                });
}
//逛商品
function autoProductTask(){
        for(let i = 0,leng = productList.length;i&lt;leng;i++){
                (function(index){
                        setTimeout(()=&gt;{
                                let item = productList[index];
                                autoPost(item['id'],4);
                                console.log(`商品总任务数：${leng} 当前任务数：${index + 1}`);
                                if( leng-1 == index){
                                        autoShopTask();
                                }
                        },index*1500)
                })(i)
        }
}
//逛店铺
function autoShopTask(){
        for(let i = 0,leng = shopList.length;i&lt;leng;i++){
                (function(index){
                        setTimeout(()=&gt;{
                                let item = shopList[index];
                                autoPost(item['id'],2);
                                console.log(`商铺总任务数：${leng} 当前任务数：${index + 1}`);
                                if( leng-1 == index){
                                        autoPlay();
                                }
                        },index*1500)
                })(i)
        }
}
//好玩互动
function autoPlay(){
        for(let i = 0,leng = 4;i&lt;leng;i++){
                (function(index){
                        setTimeout(()=&gt;{
                                autoPost(0,5);
                                console.log(`好玩互动：${leng} 当前任务数：${index + 1}`);
                                if( leng-1 == index){
                                        autoInteract();
                                }
                        },index*1000)
                })(i)
        }
}
//视频直播
function autoInteract(){
        for(let i = 0,leng = 4;i&lt;leng;i++){
                (function(index){
                        setTimeout(()=&gt;{
                                autoPost(0,10);
                                console.log(`视频直播：${leng} 当前任务数：${index + 1}`);
                                if( leng-1 == index){
                                        autoShopping();
                                }
                        },index*1000)
                })(i)
        }
}
//精彩会场
function autoShopping(){
        for(let i = 0,leng = 3;i&lt;leng;i++){
                (function(index){
                        setTimeout(()=&gt;{
                                autoPost(0,3);
                                console.log(`精彩会场：${leng} 当前任务数：${index + 1}`);
                                },
                        index*1000)
                })(i)
        }
}
start();
