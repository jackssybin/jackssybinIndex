<h2>本章概述</h2>
<p>Spring Boot 4 在事务管理方面带来了改进，特别是在虚拟线程环境下的事务处理、响应式事务增强等。</p>
<p><strong>本章重点</strong>:</p>
<ul>
<li>✅ 事务传播行为的变化</li>
<li>✅ 虚拟线程下的事务管理</li>
<li>✅ 响应式事务增强</li>
<li>✅ 分布式事务解决方案</li>
<li>✅ 与 Spring Boot 3 的对比</li>
</ul>
<h2>7.1 事务传播行为的变化</h2>
<h3>7.1.1 事务基础回顾</h3>
<p>Spring 支持7种事务传播行为：</p>
<table>
<thead>
<tr>
<th>传播行为</th>
<th>说明</th>
<th>Spring Boot 4 变化</th>
</tr>
</thead>
<tbody>
<tr>
<td>REQUIRED</td>
<td>需要事务，没有则创建</td>
<td>性能优化</td>
</tr>
<tr>
<td>REQUIRES_NEW</td>
<td>总是创建新事务</td>
<td>虚拟线程优化</td>
</tr>
<tr>
<td>SUPPORTS</td>
<td>支持事务，没有也可以</td>
<td>无变化</td>
</tr>
<tr>
<td>NOT_SUPPORTED</td>
<td>不支持事务</td>
<td>无变化</td>
</tr>
<tr>
<td>MANDATORY</td>
<td>必须在事务中</td>
<td>无变化</td>
</tr>
<tr>
<td>NEVER</td>
<td>不能在事务中</td>
<td>无变化</td>
</tr>
<tr>
<td>NESTED</td>
<td>嵌套事务</td>
<td>改进的实现</td>
</tr>
</tbody>
</table>

<h4>3. 服务层</h4>
<p><strong>AccountService.java</strong>:</p>
<pre><code class="language-java">package com.example.transaction.service;
<p>import com.example.transaction.entity.Account;
import com.example.transaction.repository.AccountRepository;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Propagation;
import org.springframework.transaction.annotation.Transactional;</p>
<p>import java.math.BigDecimal;</p>
<p>@Service
public class AccountService {
private final AccountRepository accountRepository;</p>
<pre><code>public AccountService(AccountRepository accountRepository) {
    this.accountRepository = accountRepository;
}
<p>/**</p>
<ul>
<li>
<p>REQUIRED - 默认传播行为
*/
@Transactional
public void debit(String accountNumber, BigDecimal amount) {
Account account = accountRepository.findByAccountNumber(accountNumber)
.orElseThrow(() -&amp;gt; new RuntimeException(&amp;quot;Account not found: &amp;quot; + accountNumber));</p>
<p>if (account.getBalance().compareTo(amount) &amp;lt; 0) {
throw new RuntimeException(&amp;quot;Insufficient balance&amp;quot;);
}</p>
<p>account.setBalance(account.getBalance().subtract(amount));
accountRepository.save(account);
}</p>
</li>
</ul>
<p>/**</p>
<ul>
<li>
<p>REQUIRED - 默认传播行为
*/
@Transactional
public void credit(String accountNumber, BigDecimal amount) {
Account account = accountRepository.findByAccountNumber(accountNumber)
.orElseThrow(() -&amp;gt; new RuntimeException(&amp;quot;Account not found: &amp;quot; + accountNumber));</p>
<p>account.setBalance(account.getBalance().add(amount));
accountRepository.save(account);
}</p>
</li>
</ul>

</ul>

<p>if (account.getBalance().compareTo(amount) &amp;lt; 0) {
throw new RuntimeException(&amp;quot;Insufficient balance&amp;quot;);
}</p>
<p>account.setBalance(account.getBalance().subtract(amount));
accountRepository.save(account);
}

</li>
</ul>

<p><strong>TransferService.java</strong>:</p>
<pre><code class="language-java">package com.example.transaction.service;
<p>import com.example.transaction.entity.TransactionLog.TransactionStatus;

<p>import java.math.BigDecimal;</p>
<p>/**</p>
<ul>

<li>

<p>/**</p>
<ul>

public void transfer(String fromAccount, String toAccount, BigDecimal amount) {

</code></pre></p>
<p>} catch (Exception e) {

<pre><code> // 记录失败日志（独立事务，即使主事务回滚也会保存）

     TransactionStatus.FAILED, e.getMessage());
<p>throw new RuntimeException(&amp;quot;Transfer failed: &amp;quot; + e.getMessage(), e);
</code></pre></p>
<p>}
}</p>

<ul>
<li>响应式转账
*/
public Mono&amp;lt;Void&amp;gt; transfer(String fromAccount, String toAccount, BigDecimal amount) {
return accountRepository.findByAccountNumber(fromAccount)
.flatMap(from -&amp;gt; {
if (from.getBalance().compareTo(amount) &amp;lt; 0) {
return Mono.error(new RuntimeException(&amp;quot;Insufficient balance&amp;quot;));
}
from.setBalance(from.getBalance().subtract(amount));
return accountRepository.save(from);
})
.then(accountRepository.findByAccountNumber(toAccount))
.flatMap(to -&amp;gt; {
to.setBalance(to.getBalance().add(amount));
return accountRepository.save(to);
})

.as(transactionalOperator::transactional);  // 应用事务
}

<pre><code> try {
     // Step 1: 扣款
