<h3>15.1.2 启动优化配置</h3>
<p><strong>application.yml</strong>:</p>
<pre><code class="language-yaml">spring:
  main:
    lazy-initialization: false  # 不推荐懒加载
<p>aot:
enabled: true
processing:
optimize-configuration: true
remove-unused-autoconfig: true</p>
<p>threads:
virtual:
enabled: true</p>
<h1>JVM 参数优化</h1>
<p>java:
options: &gt;
-XX:+UseG1GC
-XX:MaxGCPauseMillis=200
-XX:+UseStringDeduplication
-Xms512m
-Xmx512m
</code></pre></p>
<h3>15.1.3 CDS (Class Data Sharing)</h3>
<p><strong>生成 CDS 归档</strong>:</p>
<pre><code class="language-bash"># 1. 生成类列表
java -Xshare:off -XX:DumpLoadedClassList=classes.lst -jar app.jar
<h1>2. 创建 CDS 归档</h1>
<p>java -Xshare:dump -XX:SharedClassListFile=classes.lst <br>
-XX:SharedArchiveFile=app.jsa -jar app.jar</p>
<h1>3. 使用 CDS 启动</h1>
<p>java -Xshare:on -XX:SharedArchiveFile=app.jsa -jar app.jar
</code></pre></p>
<p><strong>性能提升</strong>:</p>
<ul>
<li>启动时间: 减少 20-30%</li>
<li>内存占用: 减少 10-15%</li>
</ul>

<pre><code class="language-java">@Component
public class PerformanceMetrics {

public MeterBinder performanceMetrics() {

        // 虚拟线程数量

            .register(registry);

<h3>15.5.2 性能基准测试</h3>
<p><strong>JMH 基准测试</strong>:</p>
<pre><code class="language-java">@BenchmarkMode(Mode.Throughput)
@OutputTimeUnit(TimeUnit.SECONDS)
@State(Scope.Benchmark)
public class VirtualThreadBenchmark {
<pre><code>@Benchmark
public void platformThreads() throws Exception {
    ExecutorService executor = Executors.newFixedThreadPool(100);
    for (int i = 0; i &amp;lt; 10000; i++) {
        executor.submit(() -&amp;gt; {
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        });
    }
    executor.shutdown();
    executor.awaitTermination(1, TimeUnit.MINUTES);
}
<p>@Benchmark
public void virtualThreads() throws Exception {
ExecutorService executor = Executors.newVirtualThreadPerTaskExecutor();
for (int i = 0; i &amp;lt; 10000; i++) {
executor.submit(() -&amp;gt; {
try {
Thread.sleep(100);
} catch (InterruptedException e) {
Thread.currentThread().interrupt();
}
});
}
executor.shutdown();
executor.awaitTermination(1, TimeUnit.MINUTES);
}
