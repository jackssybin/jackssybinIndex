<h1>第7章：事务管理改进</h1>
<h2>本章概述</h2>
<p>Spring Boot 4 在事务管理方面带来了改进，特别是在虚拟线程环境下的事务处理、响应式事务增强等。</p>
<p><strong>本章重点</strong>:</p>
<ul>
<li>✅ 事务传播行为的变化</li>
<li>✅ 虚拟线程下的事务管理</li>
<li>✅ 响应式事务增强</li>
<li>✅ 分布式事务解决方案</li>
<li>✅ 与 Spring Boot 3 的对比</li>
</ul>
<h2>7.1 事务传播行为的变化</h2>
<h3>7.1.1 事务基础回顾</h3>
<p>Spring 支持7种事务传播行为：</p>
<table>
<thead>
<tr>
<th>传播行为</th>
<th>说明</th>
<th>Spring Boot 4 变化</th>
</tr>
</thead>
<tbody>
<tr>
<td>REQUIRED</td>
<td>需要事务，没有则创建</td>
<td>性能优化</td>
</tr>
<tr>
<td>REQUIRES_NEW</td>
<td>总是创建新事务</td>
<td>虚拟线程优化</td>
</tr>
<tr>
<td>SUPPORTS</td>
<td>支持事务，没有也可以</td>
<td>无变化</td>
</tr>
<tr>
<td>NOT_SUPPORTED</td>
<td>不支持事务</td>
<td>无变化</td>
</tr>
<tr>
<td>MANDATORY</td>
<td>必须在事务中</td>
<td>无变化</td>
</tr>
<tr>
<td>NEVER</td>
<td>不能在事务中</td>
<td>无变化</td>
</tr>
<tr>
<td>NESTED</td>
<td>嵌套事务</td>
<td>改进的实现</td>
</tr>
</tbody>
</table>

<h2>7.3 分布式事务解决方案</h2>
<h3>7.3.1 Saga 模式</h3>
<p><strong>Saga 编排器</strong>:</p>
<pre><code class="language-java">package com.example.transaction.saga;
<p>import org.springframework.stereotype.Service;</p>
<p>import java.math.BigDecimal;
import java.util.ArrayList;
import java.util.List;</p>
<p>/**</p>
<ul>
<li>
<p>Saga 模式 - 分布式事务
*/
@Service
public class TransferSaga {</p>
<p>private final AccountService accountService;
private final TransactionLogService logService;</p>
<p>public TransferSaga(AccountService accountService,
TransactionLogService logService) {
this.accountService = accountService;
this.logService = logService;
}</p>
<p>public void executeTransfer(String fromAccount, String toAccount, BigDecimal amount) {
List&lt;SagaStep&gt; executedSteps = new ArrayList&lt;&gt;();</p>
<pre><code> try {
     // Step 1: 扣款
     SagaStep debitStep = new SagaStep(
         &amp;quot;debit&amp;quot;,
         () -&amp;gt; accountService.debit(fromAccount, amount),
         () -&amp;gt; accountService.credit(fromAccount, amount)  // 补偿操作
     );
     debitStep.execute();
     executedSteps.add(debitStep);
<pre><code> // Step 2: 入账
 SagaStep creditStep = new SagaStep(
     &amp;amp;quot;credit&amp;amp;quot;,
     () -&amp;amp;gt; accountService.credit(toAccount, amount),
     () -&amp;amp;gt; accountService.debit(toAccount, amount)  // 补偿操作
 );
 creditStep.execute();
 executedSteps.add(creditStep);
 
 // Step 3: 记录日志
 SagaStep logStep = new SagaStep(
     &amp;amp;quot;log&amp;amp;quot;,
     () -&amp;amp;gt; logService.logTransaction(fromAccount, toAccount, amount, 
         TransactionLog.TransactionStatus.SUCCESS, null),
     () -&amp;amp;gt; {}  // 日志不需要补偿
 );
 logStep.execute();
 executedSteps.add(logStep);
</code></pre>
<p>} catch (Exception e) {
// 执行补偿操作
compensate(executedSteps);
throw new RuntimeException(&amp;quot;Transfer failed, compensated&amp;quot;, e);
}
</code></pre></p>
<p>}</p>
<p>private void compensate(List&lt;SagaStep&gt; executedSteps) {
// 逆序执行补偿操作
for (int i = executedSteps.size() - 1; i &gt;= 0; i--) {
try {
executedSteps.get(i).compensate();
} catch (Exception e) {
// 记录补偿失败
System.err.println(&quot;Compensation failed for step: &quot; +
executedSteps.get(i).name);
}
}
}</p>

 }
